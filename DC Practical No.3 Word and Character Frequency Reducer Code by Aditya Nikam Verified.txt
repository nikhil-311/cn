Title: Design a distributed application using MapReduce under Hadoop for: a) Character 
counting in a given text file. b) Counting no. of occurrences of every word in a given text 
file. 

#!/usr/bin/env python3
import sys

current_char = None
current_count = 0

for line in sys.stdin:
    char, count = line.strip().split('\t', 1)

    try:
        count = int(count)
    except ValueError:
        continue

    if current_char == char:
        current_count += count
    else:
        if current_char:
            print(f"{current_char}\t{current_count}")
        current_char = char
        current_count = count

if current_char == char:
    print(f"{current_char}\t{current_count}")
